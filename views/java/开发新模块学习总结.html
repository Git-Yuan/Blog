    # 1.前言

    最近刚进公司实习，我的岗位本来是一个前端工程师，组长问我有没有兴趣学一学后端，我回了一句还行，过了几天就接到了这个需求，在原有项目上添加一个新的功能模块作为练手，这给原本就不富裕的家庭又雪上加霜啊。没办法，硬着头皮学了，其实在学校也学习过一些基础的数据库，对于一些增删改查的操作问题还是不大的，而且也自己也写过一些基础的接口。所以在学习了几天公司项目的代码之后，发现比我想的要更容易上手，只要能够理解这些Entity层，Dto层，Service层，Impl，action层的作用和用法，照着样子还是能够写出一些基础的查询接口来的。现在更能明白站在巨人的肩膀上是有多轻松了，苦啊累得都被巨人抗了。

    4.serviceImpl:接口实现类。

    package com.beichende.devicemanager.web.service.impl;

import com.beichende.core.base.dao.AdmbaseDao;
import com.beichende.core.base.dao.PaginationSupport;
import com.beichende.devicemanager.entity.PartTypeConfig;
import com.beichende.devicemanager.web.service.PartTypeConfigService;
import com.beichende.devicemanager.web.dto.PartTypeConfigDisplayDto;
import com.beichende.devicemanager.web.dto.PartTypeConfigQueryDto;
import java.util.*;
import org.apache.commons.beanutils.BeanUtils;
import org.apache.commons.lang.StringUtils;
import org.apache.log4j.Logger;

public class PartTypeConfigServiceImpl implements PartTypeConfigService
{

	private static final Logger logger = Logger.getLogger("com.beichende.devicemanager.web.service.impl.PartTypeNewServiceImpl");
	private AdmbaseDao admbaseDao;

	public PartTypeConfigServiceImpl()
	{
	}

	@Override
	public Map partTypeConfig(PartTypeConfigQueryDto partTypeConfigQueryDto, int startIndex, int pageSize) throws Exception {
		Map map = new HashMap();
		StringBuilder joinCond = new StringBuilder();
		joinCond.append("from PartTypeConfig where 1=1");
		String compoTypeNo = partTypeConfigQueryDto.getCompoNo();
		String compoTypeName = partTypeConfigQueryDto.getCompoName();
		if (!StringUtils.isBlank(compoTypeNo))
			joinCond.append(" and partTypeNo ='").append(compoTypeNo).append("'");
		if (!StringUtils.isBlank(compoTypeName))
			joinCond.append(" and partTypeName ='").append(compoTypeName).append("'");
		joinCond.append(" order by id desc");
		logger.info("开始分页查询部件类型信息...");
		try {
			PaginationSupport pagination = admbaseDao.findbyPage(joinCond.toString(), startIndex, pageSize);
			int total = pagination.getTotalCount();
			if (total > 0)
			{
				List list = pagination.getItems();
				int size = list.size();
				PartTypeConfig partTypeConfigEntity = null;
				List partTypeConfigDisplayDtos = new ArrayList();
				for (int i = 0; i < size; i++)
				{
					partTypeConfigEntity = (PartTypeConfig)list.get(i);
					PartTypeConfigDisplayDto partTypeConfigDisplayDto = new PartTypeConfigDisplayDto();
					BeanUtils.copyProperties(partTypeConfigDisplayDto, partTypeConfigEntity);
					partTypeConfigDisplayDtos.add(partTypeConfigDisplayDto);
					partTypeConfigDisplayDto = null;
				}
				map.put("rows", partTypeConfigDisplayDtos);
			}
			map.put("total", Integer.valueOf(total));
		}catch(Exception e){
			logger.warn((new StringBuilder()).append("serviceImpl层处理部件类型【查询】异常[").append(e.getMessage()).append("]").toString(), e);
			throw e;
		}
		return map;
	}
}

    package com.beichende.devicemanager.web.action;

import java.util.HashMap;
import java.util.Map;
import org.apache.log4j.Logger;
import org.springframework.beans.factory.annotation.Autowired;

import com.beichende.devicemanager.entity.PartTypeNew;
import com.beichende.devicemanager.web.service.PartTypeConfigService;
import com.beichende.devicemanager.web.dto.PartTypeConfigQueryDto;
import com.beichende.devicemanager.web.dto.PartTypeConfigDisplayDto;
import com.opensymphony.xwork2.ActionSupport;
import com.opensymphony.xwork2.Action;
import net.sf.json.JSONObject;

public class PartTypeConfigAction extends ActionSupport{
	private static final long serialVersionUID = 1L;
	private static final Logger logger = Logger.getLogger("com.beichende.devicemanager.web.action.PartTypeConfigAction");
	private PartTypeConfigService partTypeConfigService;
	private PartTypeConfigQueryDto partTypeConfigQueryDto;
	private String page;
	private String rows;
	private JSONObject reJSONObject;
	
	public PartTypeConfigAction()
	{
	}

	public String partTypeConfigQuery()
	{
		logger.info("进入action层部件类型查询方法...");
		Map map = new HashMap();
		int currentPage = Integer.parseInt(null != page && !"0".equals(page) ? page : "1");
		int pageSize = Integer.parseInt(null != rows && !"0".equals(rows) ? rows : "20");
		int startIndex = (currentPage - 1) * pageSize;
		try
		{
			map = partTypeConfigService.partTypeConfig(partTypeConfigQueryDto, startIndex, pageSize);
			if (null != map.get("rows"))
			{
				map.put("partTypeConfigQueryFlag", Boolean.valueOf(true));
			} else
			{
				map.put("partTypeConfigQueryFlag", Boolean.valueOf(false));
				map.put("rows", new String[0]);
			}
			reJSONObject = JSONObject.fromObject(map);
		}
		catch (Exception e)
		{
			logger.warn((new StringBuilder()).append("action层返回部件类型【查询】结果异常[").append(e.getMessage()).append("]").toString(), e);
			reJSONObject = JSONObject.fromObject("{total:0,rows:[],compoQueryExFlag:true}");
		}
		return "success";
	}
	
	public String getPage() {
		return page;
	}

	public void setPage(String page) {
		this.page = page;
	}

	public String getRows() {
		return rows;
	}

	public void setRows(String rows) {
		this.rows = rows;
	}

	public JSONObject getReJSONObject() {
		return reJSONObject;
	}

	public void setReJSONObject(JSONObject reJSONObject) {
		this.reJSONObject = reJSONObject;
	}
}
